Lab 06 - Application Security

 sudo bash -c 'echo 0 > /proc/sys/kernel/randomize_va_space'

Enable ASLR 

 sudo bash -c 'echo 2 > /proc/sys/kernel/randomize_va_space'

Disable canaries: 

 gcc overflow.c -o overflow -fno-stack-protector

Overview

A buffer overflow occurs when data written to a buffer overruns its boundary and overwrites adjacent memory locations, due to insufficient bounds checking.

Exercises

1. Simple buffer overflow

  • bufferoverflow1
#include <stdio.h>
#include <unistd.h>
#include <string.h>


void wanted(){
	puts("Well done!");
}

void copy(char *arg){
	char buf[5];
  	strcpy(buf, arg);
}

int main(int argc, char **argv) {
	copy(argv[1]); 
  	return 0;
}

2. Getting shell

  • bufferoverflow2
#include <stdio.h>
#include <unistd.h>
#include <string.h>


void wanted(){
	execve("/bin/sh", NULL, NULL);
}

void function() {
   char buffer[10];
   scanf("%s", buffer);
}

int main() {
  function();
  return 0;
}

3. Shellcode

  • shellcode
#include <stdio.h>
#include <unistd.h>
#include <string.h>


void vuln(char *str) {
   char buffer[50];
   strcpy(buffer,str);
}

int main() {
  char buf[256];
  fgets(buf, 256, stdin);
  vuln(buf);
  
  return 0;
}
