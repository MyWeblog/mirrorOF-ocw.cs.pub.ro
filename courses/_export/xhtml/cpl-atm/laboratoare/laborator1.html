Exemplu 1

%lex
%%
 
[0-9]+    return 'NUMBER'
'+'   return '+'
'-'   return '-'
 
/lex
%%
 
expr : NUMBER '+' NUMBER { alert(parseInt($1)+parseInt($3)); }
     | NUMBER '-' NUMBER { alert(parseInt($1)-parseInt($3)); }
     | NUMBER { $$ = parseInt($1); }
     ;
