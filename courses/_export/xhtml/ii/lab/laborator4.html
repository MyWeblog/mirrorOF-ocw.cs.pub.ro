    #include <stdio.h>
 
int divide(int a, int b)
{
    printf("Dividing %d by %d\n", a, b);
    return a / b;
}
 
int main(void)
{
    int x, y, result;
 
    x = 10; y = 2;
    result = divide(x, y);
    printf("%d\n", result);
 
    x = 5; y = 0;
    result = divide(x, y);
    printf("%d\n", result);
 
    return 0;
}
